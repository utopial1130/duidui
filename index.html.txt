  <script>
    const tg = window.Telegram?.WebApp;
    const textarea = document.getElementById("question");
    const submitBtn = document.getElementById("submitBtn");
    const statusEl = document.getElementById("status");
    const chips = document.querySelectorAll(".chip");

    if (tg) {
      tg.ready();
      tg.expand();

      try {
        tg.setHeaderColor("#ffe7f1");
        tg.setBackgroundColor("#fff7fb");
      } catch (e) {}

      const user = tg.initDataUnsafe?.user;
      if (user?.first_name) {
        statusEl.textContent = `你好，${user.first_name}`;
      } else {
        statusEl.textContent = "欢迎使用提问";
      }
    } else {
      statusEl.textContent = "当前是普通浏览器预览模式";
    }

    chips.forEach((chip) => {
      chip.addEventListener("click", () => {
        textarea.value = chip.dataset.text || "";
        textarea.focus();
      });
    });

    submitBtn.addEventListener("click", () => {
      const text = textarea.value.trim();

      if (!text) {
        statusEl.textContent = "先输入一个问题吧";
        textarea.focus();
        return;
      }

      const payload = {
        type: "question_submit",
        text,
        time: Date.now()
      };

      if (tg) {
        tg.sendData(JSON.stringify(payload));
        statusEl.textContent = "已发送给机器人";
      } else {
        statusEl.textContent = "预览模式：这里会把问题发送给 Telegram Bot";
        console.log("sendData:", payload);
      }
    });
  </script>
